// SUT: NewBag
package corpus

import (
	"testing"

	"github.com/stretchr/testify/assert"
)

type bag struct{ items []string }

func (b *bag) Has(s string) bool {
	for _, i := range b.items {
		if i == s {
			return true
		}
	}
	return false
}

func TestNewBag_HasItem(t *testing.T) {
	b := NewBag([]string{"x", "y"})
	assert.True(t, b.Has("x"))
}
